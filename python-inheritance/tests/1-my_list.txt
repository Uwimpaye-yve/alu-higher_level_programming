# Import and instantiate MyList
MyList = __import__('1-my_list').MyList

# Check instantiation
my_list = MyList()
print(isinstance(my_list, list))  # Expected output: True

# Check append and print
my_list.append(1)
my_list.append(4)
my_list.append(2)
print(my_list)  # Expected output: [1, 4, 2]

# Check sorted print without changing original list
my_list.print_sorted()  # Expected output: [1, 2, 4]
print(my_list)  # Expected output: [1, 4, 2]

# Test negative numbers and print_sorted with unsorted appends
my_list.append(-3)
my_list.print_sorted()  # Expected output: [-3, 1, 2, 4]

# Test empty list
empty_list = MyList()
empty_list.print_sorted()  # Expected output: []

# Test print_sorted returns a new list
print(id(my_list) != id(sorted(my_list)))  # Expected output: True
